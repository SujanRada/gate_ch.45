\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[auto, node distance=1.5cm]
    % Define blocks and styles
    \tikzstyle{block} = [draw, rectangle, minimum height=3em, minimum width=6em]
    
    % Place nodes
    \node [block] (G_c) {$G_c$};
    \node [block, right=of G_c] (G_f) {$G_f$};
    \node [block, right=of G_f] (G_p) {$G_p$};
    \node [block, below=of G_f] (G_M) {$G_M$};
    
    % Draw lines between boxes
    \draw (G_c.east) -- ++(0.3,0) coordinate (line1) |- (G_f.west);
    \draw (G_f.east) -- ++(0.3,0) coordinate (line2) |- (G_p.west);
    
    % Draw additional lines
    \draw (G_c.south) -- ++(0,-2) -| (G_M.west);
    \draw (G_p.south) -- ++(0,-2) -| (G_M.east);
    
\end{tikzpicture}
